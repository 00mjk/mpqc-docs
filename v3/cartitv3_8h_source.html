36  public:
37  CartesianIterV3(int l) : CartesianIter(l) {}

39  void start() {
40  bfn_=a_=c_=0;
41  b_=l_;
42  }
43 
44  void next() {
45  if (c_<l_-a_)
46  c_++;
47  else {
48  c_=0;
49  a_++;
50  }
51  bfn_++;
52  b_ = l_-a_-c_;
53  }
54 
55  operator int() {
