35 class CartesianIterCCA : public CartesianIter {

37  public:
38  CartesianIterCCA(int l) : CartesianIter(l) {}

40  void start() {
41  bfn_=b_=c_=0;
42  a_=l_;
43  }
44 
45  void next() {
46  if (c_ < l_ - a_) {
47  b_--;
48  c_++;
49  }
50  else {
51  a_--;
52  c_ = 0;
53  b_ = l_ - a_;
54  }
55  bfn_++;
56  }

58  operator int() {
59  return (a_ >= 0);
60  }

67  int bfn() {
68  int i = a();
69  int am = l();
70  if (am == i)
71  return 0;
72  else {
73  int j = b();
74  int c = am - i;
