183  struct IntDescrFactory {
184  template <int NumCenters>
185  static Ref< typename NCentersToIntDescr<NumCenters,2>::value >
186  make(const Ref<Integral>& integral,
187  TwoBodyOperSet::type type,
188  const Ref<IntParams>& params) {
189  typedef typename NCentersToIntDescr<NumCenters,2>::value ReturnType;
190  switch (type) {
191  case TwoBodyOperSet::ERI: {
192  typedef TwoBodyNCenterIntDescr<NumCenters,TwoBodyOperSet::ERI> ConcreteType;
193  return new ConcreteType(integral,params);
194  } break;
195  case TwoBodyOperSet::R12: {
196  typedef TwoBodyNCenterIntDescr<NumCenters,TwoBodyOperSet::R12> ConcreteType;
197  return new ConcreteType(integral,params);
198  } break;
199  case TwoBodyOperSet::G12: {
200  typedef TwoBodyNCenterIntDescr<NumCenters,TwoBodyOperSet::G12> ConcreteType;
201  return new ConcreteType(integral,params);
202  } break;
203  case TwoBodyOperSet::G12NC: {
204  typedef TwoBodyNCenterIntDescr<NumCenters,TwoBodyOperSet::G12NC> ConcreteType;
205  return new ConcreteType(integral,params);
206  } break;
207  case TwoBodyOperSet::G12DKH: {
208  typedef TwoBodyNCenterIntDescr<NumCenters,TwoBodyOperSet::G12DKH> ConcreteType;
209  return new ConcreteType(integral,params);
210  } break;
211  case TwoBodyOperSet::R12_0_G12: {
212  typedef TwoBodyNCenterIntDescr<NumCenters,TwoBodyOperSet::R12_0_G12> ConcreteType;
213  return new ConcreteType(integral,params);
214  } break;
215  case TwoBodyOperSet::R12_m1_G12: {
216  typedef TwoBodyNCenterIntDescr<NumCenters,TwoBodyOperSet::R12_m1_G12> ConcreteType;
217  return new ConcreteType(integral,params);
218  } break;
219  case TwoBodyOperSet::G12_T1_G12: {
220  typedef TwoBodyNCenterIntDescr<NumCenters,TwoBodyOperSet::G12_T1_G12> ConcreteType;
221  return new ConcreteType(integral,params);
222  } break;
223  case TwoBodyOperSet::DeltaFunction : {
224  typedef TwoBodyNCenterIntDescr<NumCenters,TwoBodyOperSet::DeltaFunction> ConcreteType;
225  return new ConcreteType(integral,params);
226  } break;
227  default:
228  MPQC_ASSERT(false);
229  }
230  return Ref< typename NCentersToIntDescr<NumCenters,2>::value >(); // dummy return statement to pacify picky compilers
231  }
232  };
