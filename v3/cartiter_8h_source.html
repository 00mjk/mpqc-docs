117 inline void
118 RedundantCartesianIter::start()
119 {
120  if (l_==0)
121  done_ = 1;
122  else
123  done_ = 0;
124 
125  for (int i=0; i<l_; i++)
126  axis_[i] = 0;
127 }
128 
129 inline void
130 RedundantCartesianIter::next()
131 {
132  for (int i=0; i<l_; i++) {
133  if (axis_[i] == 2)
134  axis_[i] = 0;
135  else {
136  axis_[i]++;
137  return;
138  }
139  }
140  done_ = 1;
141 }

143 inline int
144 RedundantCartesianIter::l(int axis)
145 {
146  int i;
147  int r = 0;
148  for (i=0; i<l_; i++) if (axis_[i]==axis) r++;
149  return r;
150 }
