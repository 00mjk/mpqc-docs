35 class CartesianIterCints : public CartesianIter {

38  public:
39  CartesianIterCints(int l) : CartesianIter(l) {}

41  void start() {
42  bfn_=b_=c_=0;
43  a_=l_;
44  }
45 
46  void next() {
47  if (c_ < l_ - a_) {
48  b_--;
49  c_++;
50  }
51  else {
52  a_--;
53  c_ = 0;
54  b_ = l_ - a_;
55  }
56  bfn_++;
57  }
58 
59  operator int() {
60  return (a_ >= 0);
61  }

64 class RedundantCartesianIterCints : public RedundantCartesianIter {
65  public:
66  RedundantCartesianIterCints(int l) : RedundantCartesianIter(l) {}
67 
68  int bfn() {
69  int i = a();
70  int am = l();
71  if (am == i)
72  return 0;
73  else {
74  int j = b();
75  int c = am - i;
76  return ((((c+1)*c)>>1)+c-j);
77  }
78  }
