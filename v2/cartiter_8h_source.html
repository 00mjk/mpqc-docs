115 inline void
116 RedundantCartesianIter::start()
117 {
118  if (l_==0)
119  done_ = 1;
120  else
121  done_ = 0;
122 
123  for (int i=0; i<l_; i++)
124  axis_[i] = 0;
125 }

127 inline void
128 RedundantCartesianIter::next()
129 {
130  for (int i=0; i<l_; i++) {
131  if (axis_[i] == 2)
132  axis_[i] = 0;
133  else {
134  axis_[i]++;
135  return;
136  }
137  }
138  done_ = 1;
139 }

141 inline int
142 RedundantCartesianIter::l(int axis)
143 {
144  int i;
145  int r = 0;
146  for (i=0; i<l_; i++) if (axis_[i]==axis) r++;
147  return r;
148 }
